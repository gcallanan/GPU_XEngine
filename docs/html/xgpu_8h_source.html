142 // Initialize the XGPU.
143 //
144 // In addition to allocating device memory and initializing private internal
145 // context, this routine calls xgpuSetHostInputBuffer() and
146 // xgpuSetHostOutputBuffer(). Be sure to set the context's array_h and
147 // matrix_h fields and the corresponding length fields) accordingly.
148 //
149 // If context->array_h is zero, an array of ComplexInput elements is allocated
150 // (of the appropriate size) via CudaMallocHost, otherwise the memory region
151 // pointed to by context->array_h of length context->array_len is registered
152 // with CUDA via the CudaHostRegister function.
153 //
154 // If context->matrix_h is zero, an array of Complex elements is allocated (of
155 // the appropriate size) via CudaMallocHost, otherwise the memory region
156 // pointed to by context->matrix_h of length context->matrix_len if registered
157 // with CUDA via the CudaHostRegister function.
158 //
159 // The index of the GPU device should be put into device_flags. In addition,
160 // the following optional flags can be or'd into this value:
161 // XGPU_DONT_REGISTER_ARRAY Disables registering (pinning) of host array
162 // XGPU_DONT_REGISTER_MATRIX Disables registering (pinning) of host matrix
163 // XGPU_DONT_REGISTER Disables registering (pinning) of all host mem
164 // E.g., xgpuInit(&ctx, device_idx | XGPU_DONT_REGISTER_ARRAY);
165 //
166 // Note that if registering is disabled, the corresponding xgpuSetHost*Buffer
167 // function _must_ be called prior to calling xgpuCudaXengine.
168 int xgpuInit(XGPUContext *context, int device_flags);
